import { Meta, Heading, ColorPalette, ColorItem } from '@storybook/addon-docs';
import tokens from './design-tokens.json';

<Meta title="Design System/Colors" />

export const generateColorsArray = (obj) => {
  return (Object.values(obj).map(val => val.value));
};

  <>
    {
      Object.entries(tokens.colors).map((token, i) => {
        return <div key={i} style={{ textTransform: "capitalize" }}>
          <Heading>
            <label style={{ textTransform: "uppercase" }}>{token[0]}</label>
          </Heading>
          <ColorPalette>
            {
              Object.entries(token[1]).map((nestedToken, j) => {
                return <ColorItem key={j}
                                  title={nestedToken[0]}
                                  subtitle=""
                                  colors={generateColorsArray(nestedToken[1])}
                />
              })
            }
          </ColorPalette>
        </div>
      })
    }
  </>
